00056 /* EBNF of a url from http://www.ietf.org/rfc/rfc2396.txt:
00057  *
00058  * alpha    = lowalpha | upalpha
00059  *
00060  *    lowalpha = "a" | "b" | "c" | "d" | "e" | "f" | "g" | "h" | "i" |
00061  *               "j" | "k" | "l" | "m" | "n" | "o" | "p" | "q" | "r" |
00062  *               "s" | "t" | "u" | "v" | "w" | "x" | "y" | "z"
00063  *
00064  *    upalpha  = "A" | "B" | "C" | "D" | "E" | "F" | "G" | "H" | "I" |
00065  *               "J" | "K" | "L" | "M" | "N" | "O" | "P" | "Q" | "R" |
00066  *               "S" | "T" | "U" | "V" | "W" | "X" | "Y" | "Z"
00067  *
00068  *    digit    = "0" | "1" | "2" | "3" | "4" | "5" | "6" | "7" |
00069  *               "8" | "9"
00070  *
00071  *    alphanum = alpha | digit
00072  *
00073  *    uric          = reserved | unreserved | escaped
00074  *
00075  *    reserved    = ";" | "/" | "?" | ":" | "@" | "&" | "=" | "+" |
00076  *                  "$" | ","
00077  *
00078  *    unreserved  = alphanum | mark
00079  *
00080  *    mark        = "-" | "_" | "." | "!" | "~" | "*" | "'" | "(" | ")"
00081  *
00082  *    escaped     = "%" hex hex
00083  *    hex         = digit | "A" | "B" | "C" | "D" | "E" | "F" |
00084  *                          "a" | "b" | "c" | "d" | "e" | "f"
00085  *
00086  *    control     = <US-ASCII coded characters 00-1F and 7F hexadecimal>
00087  *    space       = <US-ASCII coded character 20 hexadecimal>
00088  *    delims      = "<" | ">" | "#" | "%" | <">
00089  *    unwise      = "{" | "}" | "|" | "\" | "^" | "[" | "]" | "`"
00090  *
00091  *  absoluteURI   = scheme ":" ( hier_part | opaque_part )
00092  *  hier_part     = ( net_path | abs_path ) [ "?" query ]
00093  *  net_path      = "//" authority [ abs_path ]
00094  *  abs_path      = "/"  path_segments
00095  *  opaque_part   = uric_no_slash *uric
00096  *
00097  *  uric_no_slash = unreserved | escaped | ";" | "?" | ":" | "@" |
00098  *                  "&" | "=" | "+" | "$" | ","
00099  *  scheme        = alpha *( alpha | digit | "+" | "-" | "." )
00100  *  authority     = server | reg_name
00101  *  reg_name      = 1*( unreserved | escaped | "$" | "," |
00102  *                  ";" | ":" | "@" | "&" | "=" | "+" )
00103  *  server        = [ [ userinfo "@" ] hostport ]
00104  *  userinfo      = *( unreserved | escaped |
00105  *                  ";" | ":" | "&" | "=" | "+" | "$" | "," )
00106  *  hostport      = host [ ":" port ]
00107  *  host          = hostname | IPv4address
00108  *  hostname      = *( domainlabel "." ) toplabel [ "." ]

00110  *  toplabel      = alpha | alpha *( alphanum | "-" ) alphanum
00111  *  IPv4address   = 1*digit "." 1*digit "." 1*digit "." 1*digit
00112  *  port          = *digit
00113  *
00114  *  path          = [ abs_path | opaque_part ]
00115  *
00116  *  path_segments = segment *( "/" segment )
00117  *  segment       = *pchar *( ";" param )
00118  *  param         = *pchar
00119  *
00120  *  pchar         = unreserved | escaped |
00121  *                  ":" | "@" | "&" | "=" | "+" | "$" | ","
