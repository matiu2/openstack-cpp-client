00066     void readRest(ostream& destination, long toReadTotal=-1) {
00067         if (toReadTotal == 0)
00068             return;
00069         // Write whatever content we already have into the destination
00070         unsigned int toRead = toReadTotal - _buffer.size();
00071         if (_buffer.size() > 0)
00072            destination << &_buffer;
00073         boost::system::error_code error;
00074         // Read until EOF or toRead, writing data to output as we go.
00075         while (boost::asio::read(_sslStream, _buffer, boost::asio::transfer_at_least(1), error)) {
00076             toRead -= _buffer.size();
00077             destination << &_buffer;
00078             if ((toRead == 0) && (toReadTotal > 0))
00079                 return;
00080         }
00081         if (error != boost::asio::error::eof)
00082             throw boost::system::system_error(error);
